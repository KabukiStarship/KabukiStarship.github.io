33 /* An ASCII List header.
34 Like most ASCII OBJ Types, the size may only be 16-bit, 32-bit, or
35 64-bit. The unsigned value must be twice the width of the signed value.
36 
37 @code
38 +==========================+ -----------
39 |_______ Buffer | ^ ^
40 |_______ ... | | |
41 |_______ Data N | Data |
42 |_______ ... | | |
43 |_______ Data 0 | v |
44 |==========================| ----- |
45 |_______ count_max | ^ |
46 |_______ ... | | |
47 |_______ Data Offset N | | |
48 |_______ ... | | Size
49 | Data Offset 1 | | |
50 |==========================| Header |
51 |_______ count_max | | |
52 |_______ ... | | |
53 |_______ Type UI1 N | | |
54 |_______ ... | | |
55 | Type UI1 1 | | | ^ Up in addresses
56 |==========================| | | |
57 | AsciiList<UI, SI> Struct | v v ^
58 +==========================+ ----------- ^ 0xN
59 @endcode
60 */
61 template <typename UI = UI4, typename SI = SI2>
62 struct CList {
63  UI size;
64  SI count_max, count;
65 };

73 /* Returns the min size of an ASCII List with the given count_max.
74 The min size is defined as enough memory to store the given count_max with
75 the largest_expected_type.
76 */
77 template <typename UI = UI4, typename SI = SI2,
78  size_t largest_expected_type = sizeof(SIW)>
79 constexpr UI ListSizeMin(SI count_max) {
80  return (UI)sizeof(CList<UI, SI>) +
81  (UI)(count_max * (largest_expected_type + sizeof(UI) + 1));
82  // << 2 to * 4.
83 }

85 /* Deletes the list contents by overwriting it with zeros. */
86 template <typename UI = UI4, typename SI = SI2>
87 void ListWipe(CList<UI, SI>* list) {
88  ASSERT(list)
89  list->count = 0;
90  UI size = list->size - sizeof(CList<UI, SI>);
91  memset(reinterpret_cast<char*>(list) + sizeof(CList<UI, SI>), 0, size);
92 }

94 /* Initializes a AsciiList from preallocated memory.
95 count_max must be in multiples of 4. Given there is a fixed size, both the
96 count_max and size will be downsized to a multiple of 4 automatically. */
97 template <typename UI = UI4, typename SI = SI2>
98 CList<UI, SI>* ListInit(UIW* buffer, UI size, SI count_max) {
99  if (!buffer) // This may be nullptr if ListNew<UI,SI> (SI, UI) failed.
100  return nullptr;
101  PRINTF("\n Initializing List with size_bytes:%u and count_max:%i",
102  (uint)size, (int)count_max)
103  SI count_max_bounds_lower = ListCountMaxBoundsLower<UI, SI>();
104  if (count_max < count_max_bounds_lower) {
105  PRINTF("\n count_max == 0 and is now %i", (int)count_max_bounds_lower)
106  count_max = count_max_bounds_lower;
107  } else {
108  PRINTF("\ncount_max was %i ", (int)count_max)
109  count_max = AlignUp<SI>(count_max);
110  PRINTF("\n and now is %i.", (int)count_max)
111  }
112 
113  // count_max = AlignUp<SI> (count_max);
114  // PRINTF ("\n Aligning up to count_max:%i", (int)count_max)
115 
116  CList<UI, SI>* list = reinterpret_cast<CList<UI, SI>*>(buffer);
117  list->size = size;
118  list->count = 0;
119  list->count_max = count_max;
120  // WIPE
121  ListWipe<UI, SI>(list);
122  return list;
123 }

125 /* Creates a list from dynamic memory. */
126 template <typename UI = UI4, typename SI = SI2>
127 UIW* ListNew(SI count_max, UI size) {
128  count_max = AlignUpUnsigned<UI8, UI>(count_max);
129  if (size < ListSizeMin<UI, SI>(count_max)) return nullptr;
130  UIW* buffer = new UIW[size >> kWordBitCount];
131 
132  CList<UI, SI>* list = reinterpret_cast<CList<UI, SI>*>(buffer);
133  list->size = size;
134  list->count = 0;
135  list->count_max = count_max;
136  // WIPE
137  ListWipe<UI, SI>(list);
138  return buffer;
139 }

201 /* Returns the last UI1 in the data array. */
202 template <typename UI = UI4, typename SI = SI2>
203 inline char* ListDataStop(CList<UI, SI>* list, SI index = -1) {
204  ASSERT(list)
205  SI count = list->count;
206  if (count == 0) {
207  if (index != -1) return nullptr;

211  UI offset = ListOffsets<UI, SI>(list)[index];

217 template <typename UI = UI4, typename SI = SI2>
218 void ListDataSpaceBelow(CList<UI, SI>* list, SI index, Socket& free_space) {
219  ASSERT(list)
220  char* data_stop;
221  if (index == 0) {
222  data_stop = ListDataBegin<UI, SI>(list);
223  free_space.begin = free_space.end = data_stop;
224  return;
225  }
226  ASSERT(index >= 0 && index <= list->count)
227  if (index == list->count) {
228  free_space.begin = ListDataStop<UI, SI>(list);
229  free_space.end = ListDataEnd<UI, SI>(list);
230  return;
231  }
232  data_stop = ListDataStop<UI, SI>(list, index);
233  UI* offsets = ListOffsets<UI, SI>(list);
234  UI offset_low = offsets[index], offset_high = offsets[index + 1];
235 }

237 /* Insets the given type-value tuple.
238  @return -1 upon failure or the index upon success. */
239 template <typename UI = UI4, typename SI = SI2>
240 SI ListInsert(CList<UI, SI>* list, type_t type, const void* value, SI index) {
241  ASSERT(list)
242  ASSERT(value)
243  PRINTF("\nInserting type:")
244  PRINT_TYPE(type, value)
245  PRINTF(" into index:%i", index)
246 
247  SI count = list->count, count_max = list->count_max;
248  if (count >= count_max || index > count || !TypeIsValid(type) || index < 0) {
249  PRINTF("\nError inserting type:%s into index %i", TypeString(type),
250  (int)index);
251  return -1;
252  }
253  PRINTF(" when count is %i", (int)count)
254 
255  type_t* types = ListTypes<UI, SI>(list);
256 
257  // 1. Check for stack push operation.
258  if (index == count) {
259  PRINTF("\nPushing element...")
260  // Push type onto the top of the type stack.
261  types[index] = type;
262  // Push the offset onto the top of the offset stack.
263  char* data_stop = ListDataStop<UI, SI>(list, count - 1);
264  PRINTF("\n Aligning data_stop from %i to ", (int)Size(list, data_stop))
265  data_stop = TypeAlignUpPointer<char>(data_stop, type);
266  PRINTF("%i", (int)Size(list, data_stop))
267  UI stop_offset = (UI)(data_stop - reinterpret_cast<char*>(list));
268  ListOffsets<UI, SI>(list)[index] = stop_offset;
269  // Write the value to the top of the value stack.
270  PRINTF(" leaving %i bytes.", (int)(ListDataEnd<UI, SI>(list) - data_stop))
271  if (!Write(data_stop, ListDataEnd<UI, SI>(list), type, value)) return -2;
272  list->count = count + 1;
273  return index;
274  }
275 
276  // 2. Shift up the types.
277  StackInsert<type_t, SI>(types, count, type, index);
278 
279  // 3. Calculate the offset to insert at.
280  char* aligned_begin = ListDataStop<UI, SI>(list, index);
281  PRINTF("\nListDataStop<UI, SI> (list) starts as %p then is aligned to ",
282  aligned_begin)
283  aligned_begin = TypeAlignUpPointer<char>(aligned_begin, type);
284  PRINTF("%p", aligned_begin)
285 
286  // 4. Insert the offset.
287  PRINTF("\nInserting into ")
288  StackInsert<UI, SI>(ListOffsets<UI, SI>(list), count, type, index);
289 
290  UI space_needed = ObjSize<UI>(value, type);
291 
292  // 5. Check if there is enough room to insert the value without shifting up
293  // any data.
294  Socket free_space;
295  ListDataSpaceBelow<UI, SI>(list, index, free_space);
296  if (Write(free_space.begin, free_space.end, type, value)) return index;
297 
298  // 6. Shift up the data enough to fit the new value.
299 
300  // 5. Write the value to the data socket.
301  if (!Write(aligned_begin, ListDataEnd<UI, SI>(list), type, value)) return -1;
302  list->count = ++count;
303  return count;
304 }

361 /* Finds a tuple that contains the given pointer. */
362 template <typename UI = UI4, typename SI = SI2>
363 SI ListFind(CList<UI, SI>* list, void* adress) {
364  ASSERT(list)
365  UI *offsets = ListOffsets<UI, SI>(list), *offset_end = offsets + list->count;
366  while (offsets < offset_end) {
367  char *begin = reinterpret_cast<char*>(list) + *offsets++,
368  *end = ListDataStop<UI, SI>(list, index);
369  if (reinterpret_cast<char*>(address) >= begin &&
370  reinterpret_cast<char*>(address) <= end)
371  return true;
372  }
373  return -1;
374 }
